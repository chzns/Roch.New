public static void CheckAndCreateTemplateFileOnDesktop(string templateContent)
    {
        // Get the path to the current user's desktop folder
        string desktopPath = Environment.GetFolderPath(Environment.SpecialFolder.Desktop);

        // Combine the desktop path with the file name "mb.txt"
        string filePath = Path.Combine(desktopPath, "mb.txt");

        // Check if the file exists
        if (!File.Exists(filePath))
        {
            try
            {
                // Create the file and write the provided template content
                File.WriteAllText(filePath, templateContent);

                Console.WriteLine("File 'mb.txt' has been created on the desktop.");
            }
            catch (Exception ex)
            {
                Console.WriteLine("An error occurred while creating the file: " + ex.Message);
            }
        }
        else
        {
            Console.WriteLine("File 'Ä£°å.txt' already exists on the desktop.");
        }
    }
